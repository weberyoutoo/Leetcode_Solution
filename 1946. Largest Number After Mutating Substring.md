class Solution {
public:
    string maximumNumber(string num, vector<int>& change) {
        int i = 0;

        for(; i < num.size(); ++i) {
            if( (num[i] - '0') < change[num[i] - '0'] )
                break;
        }

        for(; i < num.size(); ++i ) {
            if( (num[i] - '0') > change[num[i] - '0'] )
                break;
            num[i] = change[num[i] - '0'] + '0';
        }

        return num;
    }
};
