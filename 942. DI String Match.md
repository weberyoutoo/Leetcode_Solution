class Solution {
public:
    vector<int> diStringMatch(string s) {
        vector<int> temp = {0};
        auto b = temp.begin();
        for(int i = 0; i<s.size(); i++){
            if(s[i]=='I'){
                temp.push_back(i+1);
                b = temp.end();
                b = next(b, -1);
            }
            if(s[i]=='D'){
                b = temp.insert(b, i+1);
            }
        }
        return temp;
    }
};
