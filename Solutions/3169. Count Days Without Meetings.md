class Solution {
public:    
    int countDays(int days, vector<vector<int>>& meetings) {
        sort(meetings.begin(), meetings.end());

        int res = 0;
        int prev_end = 0;
        for( auto &m : meetings ) {
            if( m[0] <= prev_end ) {
                prev_end = max(prev_end, m[1]);
            }
            else {
                res += m[0] - prev_end - 1;
                prev_end = m[1];
            }
        }

        return res + (days - prev_end);
    }
};
