class Solution {
private:
    static bool init;
    static long long large_thr;
public:
    Solution() {
        if( !init ) {
            init = true;
            while( large_thr < INT_MAX ) {
                large_thr *= 3;
            }
            large_thr /= 3;
        }
    }
    bool isPowerOfThree(int n) {
        if( n <= 0 ) return false;
        return large_thr % n == 0;
    }
};
bool Solution::init = false;
long long Solution::large_thr = 3;
