# Code

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode* temp = head;
        int size = 1, idx = 1;
        while(temp->next != nullptr){
            temp = temp->next;
            size++;
        }

        if(size == 1)
            return nullptr;
        if(n == size)
            return head->next;

        temp = head;
        while(idx != (size-n)){
            temp = temp->next;
            idx++;
        }
        temp->next = temp->next->next;

        return head;
    }
};
```
# Alternative Solution
```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode* temp = new ListNode(-1);
        temp->next = head;
        ListNode *post = temp, *prev = temp;
        for(int i = 0; i<=n; ++i)
            post = post->next;
        
        while(post != nullptr){
            prev = prev->next;
            post = post->next;
        }
        ListNode* tar = prev->next;
        prev->next = prev->next->next;
        delete tar;

        return temp->next;
    }
};
```
