class Solution {
public:
    int findCharIdx(string& str, char target){
        for(int i = 0; i<str.size(); i++){
            if(str[i] == target)
                return i;
        }
        return -1;
    }
    int lengthOfLongestSubstring(string s) {
        int max = 0;
        string window;
        for(char c : s){
            int idx = findCharIdx(window, c);
            if(idx==-1){
                window.append(1, c);
                max = max < window.size() ? window.size() : max;
            }
            else{
                window.erase(0, idx+1);
                window.append(1, c);
            }
        }

        return max;
    }
};
