```cpp
class Solution {
public:
    int climbStairs(int n) {
        int first = 1, second = 2, temp;
        for(int i = 3; i <= n; ++i) {
            temp = second;
            second += first;
            first = temp;
        }

        return n == 1 ? 1 : second;
    }
};
```
