class Solution {
public:
    int minPatches(vector<int>& nums, int n) {
        uint target = 1, result = 0;
        int i = 0;
        
        while( target <= n ) {
            if( i < nums.size() && target >= nums[i] ){
                target += nums[i];
                ++i;
            }
            else {
                ++result;
                target <<= 1;
            }
        }

        return result;
    }
};
