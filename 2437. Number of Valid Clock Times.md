```cpp
class Solution {
public:
    int countTime(string time) {
        int hours = 1;
        int mins = 1;
        bool q_marks[] = {false,false,false,false,false};
        int digit_time[] = {0,0,0,0,0};
        int i, j;
        for(i = 0; i<5; i++){
            if(time[i]=='?')
                q_marks[i]=true;
            else
                digit_time[i] = time[i] - 48;
        }
        if(q_marks[0]&q_marks[1])
            hours=24;
        else if(q_marks[0]){
            if(digit_time[1] < 4)
                hours=3;
            else
                hours=2;
        }
        else if(q_marks[1]){
            if(digit_time[0] < 2)
                hours=10;
            else
                hours=4;
        }

        if(q_marks[3]&q_marks[4])
            mins=60;
        else if(q_marks[3])
            mins=6;
        else if(q_marks[4])
            mins=10;
        return hours*mins;
    }
};
```
