class Solution {
public:
    int prefixCount(vector<string>& words, string pref) {
        int n = words.size(), m = pref.size(), res = 0;

        for(int i = 0; i < n; ++i) {
            if( words[i].size() >= m ) {
                res += words[i].find(pref) == 0;
            }
        }

        return res;
    }
};
