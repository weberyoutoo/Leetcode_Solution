### 1. std::priority_queue Solution
```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    struct greater_Node{
        bool operator()(const ListNode* node1,const ListNode* node2){
            return (node1->val) > (node2->val); 
        }
    };

    priority_queue<ListNode*, vector<ListNode*>, greater_Node> Priority_Q;

    inline ListNode* extract(){
        ListNode* ret = Priority_Q.top();
        Priority_Q.pop();
        if(ret->next != nullptr)
            Priority_Q.push(ret->next);
        return ret;
    }

    ListNode* mergeKLists(vector<ListNode*>& lists) {
        ListNode *merged, *head;

        for(ListNode* node : lists){
            if(node != nullptr)
                Priority_Q.push(node);
        }

        if(Priority_Q.empty())
            return nullptr;
        else{
            merged = head = extract();
        }

        while(!Priority_Q.empty()){
            merged->next = extract();
            merged = merged->next;
        }
        
        return head;
    }
};
```
### 2. Min_Heap base algorithm from scratch
```cpp
class Heap{
public:
    vector<ListNode*> nodes = {};
    
    Heap();
    Heap(vector<ListNode*> &lists){
        for(ListNode* node : lists){
            if(node!=nullptr){
                nodes.push_back(node);
            }
        }
        for(int i = (nodes.size()+1)/2-1; i >= 0; i--)
            Adjust(i);
    }

    static inline int LeftChild(int idx){
        return 2*idx+1;
    }

    static inline int RightChild(int idx){
        return 2*idx+2;
    }

    inline int size(){
        return nodes.size();
    }

    inline void Swap(int first, int second){
        ListNode* temp = nodes[first];
        nodes[first] = nodes[second];
        nodes[second] = temp;
    }

    void Adjust(int idx){
        if(LeftChild(idx) >= nodes.size())
            return;
        else if(RightChild(idx) >= nodes.size()){
            if(nodes[idx]->val > nodes[LeftChild(idx)]->val)
                Swap(idx, LeftChild(idx));
            return;
        }
        else{
            if(nodes[idx]->val > nodes[LeftChild(idx)]->val && 
                nodes[LeftChild(idx)]->val <= nodes[RightChild(idx)]->val){
                Swap(idx, LeftChild(idx));
                Adjust(LeftChild(idx));
            }
            else if(nodes[idx]->val > nodes[RightChild(idx)]->val && 
                nodes[RightChild(idx)]->val < nodes[LeftChild(idx)]->val){
                Swap(idx, RightChild(idx));
                Adjust(RightChild(idx));
            }
        }
    }

    ListNode* min(){
        if(nodes.size()==0)
            return nullptr;
        return nodes[0];
    }

    void delMin(){
        if(nodes.size() == 0)
            return;

        nodes[0] = nodes[0]->next;
        if(nodes[0] == nullptr){
            Swap(0, nodes.size()-1);
            nodes.pop_back();
        }
        Adjust(0);
    }
};
class Solution {
public:

    ListNode* mergeKLists(vector<ListNode*>& lists) {
        Heap min_heap(lists);
        ListNode *merged, *head;

        if(min_heap.min()==nullptr)
            head = merged = nullptr;
        else
            head = merged = min_heap.min();
        min_heap.delMin();
        
        while(min_heap.size()>0){
            merged->next = min_heap.min();
            min_heap.delMin();
            merged = merged->next;
        }

        return head;
    }
};
```
