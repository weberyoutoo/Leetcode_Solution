class Solution {
public:
    bool isAnagram(string s, string t) {
        vector<int> char_count(26, 0);

        for(char c : s)
            char_count[c - 'a']++;

        for(char c : t)
            char_count[c - 'a']--;
        

        for(int count : char_count) {
            if( count != 0 )
                return false;
        }

        return true;
    }
};
