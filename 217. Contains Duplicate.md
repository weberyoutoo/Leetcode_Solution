```cpp
class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        unordered_set<int> appeared_nums;
        
        for(int &num : nums) {
            if( appeared_nums.find(num) != appeared_nums.end() )
                return true;

            appeared_nums.insert(num);
        }

        return false;
    }
};
```
