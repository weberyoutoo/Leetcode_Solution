```cpp
# define MAX 100001

class Solution {
public:
    int maxTotalReward(vector<int>& rewardValues) {
        sort(rewardValues.begin(), rewardValues.end());
        bitset<MAX> dp; 

        dp.set(0);
        int shift = 0;
        int result = 0;
        for(auto val : rewardValues){
            shift = 1e5 - (val - 1);
            dp |= dp << shift >> shift << val;
            result = result + val > MAX ? MAX : result + val;
        }
        while(!dp[result])
            result--;
            
        return result;
    }
};
```
