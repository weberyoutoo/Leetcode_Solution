class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int diff;
        vector<int> out;
        unordered_map<int,int> seen;
        for(int i = 0; i < nums.size(); i++){
            diff = target - nums[i];
            if(seen.find(diff) != seen.end()){
                out = {i, seen[diff]};
                break;
            }
            else
                seen[nums[i]] = i;
        }
        return out;
    }
};
