class Solution {
public:
    string compressedString(string word) {
        int n = word.size();
        string result;

        char rep = word[0];
        int len = 1;
        
        for(int i = 1; i < n; ++i) {

            if( word[i] == rep && len < 9 ) {
                len++;
            }

            else {
                result.push_back(len + '0');
                result.push_back(rep);

                rep = word[i];
                len = 1;
            }

        }

        result.push_back(len + '0');
        result.push_back(rep);

        return result;
    }
};
